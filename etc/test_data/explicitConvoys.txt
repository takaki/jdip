VARIANT_ALL Standard

# simple basic test
CASE EC.1 Basic Test of Explicit Convoying
PRESTATE_SETPHASE Spring 1901, Movement
PRESTATE
	England: F nth
	England: A lon
ORDERS
	England: F nth C A lon-bel
	England: A lon-nth-bel
POSTSTATE
	England: F nth
	England: A bel
END

# two armies with two convoys
CASE EC.5 (DATC 6.C.6)
PRESTATE_SETPHASE Spring 1901, Movement
PRESTATE
	England: F nth
	England: A lon
	France: F eng
	France: A bel
ORDERS
	England: F nth C A lon-bel
	England: A lon-nth-bel
	France: F eng C A bel-lon
	France: A bel-eng-lon
POSTSTATE
	England: F nth
	England: A bel
	France: F eng
	France: A lon
END


# disrupted unit swap (lon-bel bounces off bur-bel)
CASE EC.6 (DATC 6.C.7)
PRESTATE_SETPHASE Spring 1901, Movement
PRESTATE
	England: F nth
	England: A lon
	France: F eng
	France: A bel
	France: A bur
ORDERS
	England: F nth C A lon-bel
	England: A lon-nth-bel
	France: F eng C A bel-lon
	France: A bel-eng-lon
	France: A bur-bel
POSTSTATE_SAME
END


# dislodge of convoying fleet on route
# poststate: per 2000 rules
# we only specify one route (ech)
#
CASE EC.7 (DATC 6.F.9)
PRESTATE_SETPHASE Spring 1901, Movement
PRESTATE
	England: F eng
	England: F nth
	England: A lon
	France: F bre
	France: F mao
ORDERS
	England: F eng C A lon-bel
	England: F nth C A lon-bel
	England: A lon-eng-bel
	France: F bre S F mao-eng
	France: F mao-eng
POSTSTATE
	England: F nth
	England: A lon
	France: F bre
	France: F eng
POSTSTATE_DISLODGED
	England: F eng
END
